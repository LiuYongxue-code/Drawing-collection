\pgfplotsset{trig format plots=rad, compat=1.11}

\begin{document}
\begin{tikzpicture}
\begin{axis}[colormap/violet, hide axis]

domain = 0:pi,

]

{sin(x)*sin(y)*(sqrt(3/(4*pi))*sin(x)*cos(y))^2},
{cos(x)*(sqrt(3/(4*pi))*sin(x)*cos(y))^2} );
\end{axis}
\end{tikzpicture}
